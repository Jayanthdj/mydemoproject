pipeline{
    agents any
    stages{
        stage('Checkout'){
            steps{
                git branch: 'main', url: 'https://github.com/Jayanthdj/mydemoproject.git'
            }
        }
        stage('Maven Validate'){
            steps{
                sh 'mvn validate'
            }
                
            }
        }
        stage('Maven Compile'){
            steps{
                sh 'mvn compile'
            }
        }
        stage('Maven Test'){
            steps{
                sh 'mvn test'
            }
        }
        stage('Maven Package'){
            steps{
                sh 'mvn package'
                }
        }
    }
